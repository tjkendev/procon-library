Link-Cut Tree

Link-Cut Tree

概要

動的木のデータ構造

Heavy-Light Decompositionを動的にしたイメージ。 分解したパスはSplay Tree上で管理する。

以下の操作ができる

  • expose(i): 根頂点から頂点iまでのパスのみを繋げる (それ以外のパスをカット

  • cut(i): 木を頂点iと頂点iの親頂点pの間の辺をカットして分割

  • link(i, p): 2つの木について、頂点iと頂点pを接続して1つの木にする

  • evert(i): 木の根頂点を頂点iに変更

計算量

  • 各クエリ: ならし \(O(\log^2 N)\)

実装(1)

各ノードの情報を別の配列で管理する実装。 各頂点は1-indexedで番号付けして管理する。

Python3が遅い。PyPy3はある程度高速。

N = 10**5
prt = [0]*(N+1)
left = [0]*(N+1)
right = [0]*(N+1)
sz = [0] + [1]*N
key = [0]*(N+1)
val = [0]*(N+1)
rev = [0]*(N+1)

def update(i, l, r):
    #assert 1 <= i <= N
    sz[i] = 1 + sz[l] + sz[r]
    val[i] = key[i] + val[l] + val[r]

def swap(i):
    if i:
        left[i], right[i] = right[i], left[i]
        rev[i] ^= 1

def prop(i):
    swap(left[i])
    swap(right[i])
    rev[i] = 0
    return 1

def splay(i):
    #assert 1 <= i <= N
    q = prt[i]
    rev[i] and prop(i)

    li = left[i]; ri = right[i]
    while q and not left[q] != i != right[q]:
        r = prt[q]
        if not r or left[r] != q != right[r]:
            if rev[q] and prop(q):
                li, ri = ri, li

                if rev[i]:
                    swap(li); swap(ri)
                    rev[i] = 0

            y = r
            if left[q] == i:
                left[q] = ri
                if ri:
                    prt[ri] = q
                update(q, ri, right[q])
                ri = q
                update(i, li, q)
            else:
                right[q] = li
                if li:
                    prt[li] = q
                update(q, left[q], li)
                li = q
                update(i, q, ri)
        else:
            rev[r] and prop(r)
            if rev[q] and prop(q):
                li, ri = ri, li

                if rev[i]:
                    swap(li); swap(ri)
                    rev[i] = 0

            y = prt[r]

            if left[r] == q:
                if left[q] == i:
                    v = left[r] = right[q]
                    if v:
                        prt[v] = r
                    update(r, v, right[r])

                    left[q] = ri; right[q] = r
                    if ri:
                        prt[ri] = q
                    update(q, ri, r)

                    prt[r] = ri = q
                    update(i, li, q)
                else:
                    left[r] = ri
                    if ri:
                        prt[ri] = r
                    update(r, ri, right[r])

                    right[q] = li
                    if li:
                        prt[li] = q
                    update(q, left[q], li)

                    ri = r; li = q
                    update(i, q, r)
            else:
                if right[q] == i:
                    v = right[r] = left[q]
                    if v:
                        prt[v] = r
                    update(r, left[r], v)

                    left[q] = r; right[q] = li
                    if li:
                        prt[li] = q
                    update(q, r, li)

                    prt[r] = li = q
                    update(i, q, ri)
                else:
                    right[r] = li
                    if li:
                        prt[li] = r
                    update(r, left[r], li)

                    left[q] = ri
                    if ri:
                        prt[ri] = q
                    update(q, ri, right[q])

                    li = r; ri = q
                    update(i, r, q)

        if y:
            if left[y] == r:
                left[y] = i
                update(y, i, right[y])
            elif right[y] == r:
                right[y] = i
                update(y, left[y], i)
        q = y
    left[i] = li; right[i] = ri
    prt[li] = prt[ri] = i
    prt[i] = q

def expose(i):
    p = 0
    cur = i
    while cur:
        splay(cur)
        right[cur] = p
        update(cur, left[cur], p)
        p = cur
        cur = prt[cur]
    splay(i)
    return i

def cut(i):
    expose(i)
    p = left[i]
    left[i] = prt[p] = 0
    return p

def link(i, p):
    expose(i)
    expose(p)
    prt[i] = p
    right[p] = i

def evert(i):
    expose(i)
    swap(i)
    rev[i] and prop(i)

def query(v):
    r = expose(v+1)
    return val[r]

def query_add(v, w):
    key[v+1] += w
    expose(v+1)

def debug(v):
    def dfs(v, k):
        if rev[v]:
            prop(v)
        if left[v]:
            dfs(left[v], k+1)
        print(" "*k, v, key[v], sz[v])
        if right[v]:
            dfs(right[v], k+1)
    print(">>>>")
    dfs(v, 0)
    print("<<<<")


readline = open(0).readline
writelines = open(1, 'w').writelines

N = int(readline())
for i in range(N):
    k, *C = map(int, readline().split())
    for c in C:
        link(c+1, i+1)
Q = int(readline())
ans = []
for q in range(Q):
    t, *args = map(int, readline().split())
    if t:
        ans.append("%d\n" % query(args[0]))
    else:
        query_add(*args)
writelines(ans)
